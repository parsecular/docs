---
title: "Cancel Order"
description: "Cancel an open order"
---

`DELETE /api/v1/orders/:order_id`

Cancels an open order on the specified exchange.

## Parameters

<ParamField path="order_id" type="string" required>
  The order identifier to cancel.
</ParamField>

<ParamField query="exchange" type="string" required>
  Exchange identifier: `polymarket`, `kalshi`, `limitless`, `opinion`, or `predictfun`.
</ParamField>

## Response

<ResponseField name="(root)" type="Order">
  <Expandable title="properties">
    <ResponseField name="id" type="string">
      Order identifier.
    </ResponseField>
    <ResponseField name="market_id" type="string">
      Market the order was on.
    </ResponseField>
    <ResponseField name="outcome" type="string">
      The outcome that was being traded.
    </ResponseField>
    <ResponseField name="side" type="string">
      `"buy"` or `"sell"`.
    </ResponseField>
    <ResponseField name="price" type="number">
      Order price.
    </ResponseField>
    <ResponseField name="size" type="number">
      Total order size.
    </ResponseField>
    <ResponseField name="filled" type="number">
      Amount filled before cancellation.
    </ResponseField>
    <ResponseField name="status" type="string">
      Will be `"cancelled"`.
    </ResponseField>
    <ResponseField name="created_at" type="string">
      ISO 8601 creation timestamp.
    </ResponseField>
    <ResponseField name="updated_at" type="string">
      ISO 8601 cancellation timestamp.
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>
```bash
curl -X DELETE \
  -H "X-API-Key: YOUR_API_KEY" \
  "https://api.parsecapi.com/api/v1/orders/order_abc123?exchange=polymarket"
```
</RequestExample>
