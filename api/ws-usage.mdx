---
title: "WebSocket Usage"
description: "Get WebSocket connection and subscription statistics"
---

`GET /api/v1/ws/usage`

Returns WebSocket usage statistics for your account or globally (admin only).

## Parameters

<ParamField query="scope" type="string" default="self">
  `self` for your own stats, `global` for all customers (admin tier required).
</ParamField>

<ParamField query="limit" type="number" default="20">
  Max results for customers and top markets lists. Range: 1–100.
</ParamField>

## Response

<ResponseField name="(root)" type="object">
  <Expandable title="properties">
    <ResponseField name="scope" type="string">
      `self` or `global`.
    </ResponseField>
    <ResponseField name="updated_at_ms" type="number">
      Unix timestamp (ms) of last update.
    </ResponseField>
    <ResponseField name="totals" type="object">
      Aggregate statistics.
      <Expandable title="properties">
        <ResponseField name="active_connections" type="number" />
        <ResponseField name="active_subscriptions" type="number" />
        <ResponseField name="connections_opened_total" type="number" />
        <ResponseField name="connections_closed_total" type="number" />
        <ResponseField name="auth_failures_total" type="number" />
        <ResponseField name="subscribe_requests_total" type="number" />
        <ResponseField name="unsubscribe_requests_total" type="number" />
        <ResponseField name="messages_sent_total" type="number" />
        <ResponseField name="bytes_sent_total" type="number" />
      </Expandable>
    </ResponseField>
    <ResponseField name="customer" type="object | null">
      Your usage (present when scope is `self`).
    </ResponseField>
    <ResponseField name="customers" type="array">
      Top customers (present when scope is `global`).
    </ResponseField>
    <ResponseField name="top_markets" type="array">
      Most subscribed markets.
      <Expandable title="properties">
        <ResponseField name="parsec_id" type="string">
          Market identifier.
        </ResponseField>
        <ResponseField name="subscriptions_total" type="number">
          Total subscriptions.
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>
```bash
# Your usage
curl -H "X-API-Key: YOUR_API_KEY" \
  "https://api.parsecapi.com/api/v1/ws/usage"

# Global usage (admin)
curl -H "X-API-Key: YOUR_ADMIN_API_KEY" \
  "https://api.parsecapi.com/api/v1/ws/usage?scope=global&limit=10"
```
</RequestExample>
